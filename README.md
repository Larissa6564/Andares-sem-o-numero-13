Usando for
for (let andar = 1; andar <= 20; andar++) {
    if (andar !== 13) {
        console.log(andar);
    }
}
usando while
let andar = 1;

while (andar <= 20) {
    if (andar !== 13) {
        console.log(andar);
    }
    andar++;
}
Usando do while
let andar = 20;

do {
    if (andar !== 13) {
        console.log(andar);
    }
    andar--;
} while (andar >= 1);
